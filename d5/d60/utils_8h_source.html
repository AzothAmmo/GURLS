00092

00097     typename std::multimap<T, T, LtCompare<T> >::iterator end = data.end();
00098

00106     unsigned long fpcumsum = 0;
00107 
00108     unsigned long idx = 0;
00109     while (it != end)
00110     {
00111         tp[idx] = static_cast<T>(gurls::gt(it->second, (T)0.0) ? tpcumsum+=1 : tpcumsum);
00112         fp[idx] = static_cast<T>(gurls::lt(it->second, (T)0.0) ? fpcumsum+=1 : fpcumsum);
00113 
00114         ++it;
00115         ++idx;
00116     }
00117 
00118     for (idx = 0; idx < N; ++idx)
00119     {
00120         rec[idx] = tp[idx]/(T)tpcumsum;
00121         prec[idx] = tp[idx]/(tp[idx]+fp[idx]);
00122     }
00123 
00124     // compute average precision
00125     T ap = 0.0;
00126     T p = 0.0;
00127     T t = 0.0;
00128 
00129     LtCompare<T> comp;

00132     const T incr = static_cast<T>(0.1);
00133 
00134     for(int steps = 0; steps<stepsNumber; ++steps)
00135     {
00136         p = 0.0;
00137         for (idx = 0; idx < N; ++idx)
00138         {
00139             if ( gurls::gt(rec[idx], t) || gurls::eq(rec[idx], t))
